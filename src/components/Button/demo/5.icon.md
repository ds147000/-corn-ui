### icon
图标按钮

```tsx
import { Button } from '@xrkmm/ui'

const Demo: React.FC = () => {

  return (
    <>
      <Button type="primary" />
      <Button type="warn" />
      <Button type="error" />
      <Button type="link" />
      <Button type="pop" />
    </>
  )
}

export default Demo
```
可以通过`icon`属性来传入图标组件实现图标按钮
